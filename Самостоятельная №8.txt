long int Sp (double a, double b, double c, int d, double n, double m){
    double p;
    if ((a<=0) || (b<=0) || (c<=0)){
        d=1;
    }
    else if ((a=b+c) || (b=a+c) || (c=a+b)){
        d=2;
    }
    else if ((a>b+c) || (b>a+c) || (c>a+b)){
        d=3;
    }
    p=(a+b+c)/2.0;
    
    n=(a+b+c);
    
    m=sqrt(p*(p-a)*(p-b)*(p-c));
    
    return d; 
}
