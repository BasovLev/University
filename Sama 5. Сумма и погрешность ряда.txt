#include <stdio.h>
#include <math.h>
int main ()
{
        int n;
	double cor, Scor, S, fcor, f, a;
	cor=0.0;
	Scor=0.0;
	S=0.0;
	a=(1.0/3.0);
	fcor=0.0;
	n=0;
	f=0.0;
	while (fabs(a-S)>0.00000001) {
	    n+=1;
	    f=(1.0/((3.0*n+1)*(3.0*n-2)));
	    fcor = f - cor;
            Scor = S + fcor;
            cor = (Scor - S) - fcor;
            S = Scor;
	}
	printf("%.8f\n", a);
	printf("%.8f\n", S);
	printf("%.8f\n", (a-S));
	printf ("%d\n", n);
	return 0;
}
