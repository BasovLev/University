#include <stdio.h>
#include <limits.h>

void prim (int m[],int size){
    int i;
    for(i=0; i<size; i++){
        printf("%3d",i+1);
    }
    printf("\n");
    for(i=0; i<size; i++){
        printf("%3d", m[i]);
    }
    printf("\n");
}

int main(){
    int a,b,c, min, i, j, n, m[100], s;
    a=0;
    b=0;
    s=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&m[i]);
    }
    min=INT_MAX;
    prim (m,n);
    printf("\n");
    for(j = 0; j < n-1; j++)
    {
        for(i=a;i<n;i++)
        {
            if (min>m[i])
            {
                min = m[i];
                c= i;
            }
            s++;
        }
       // if(m[c] != m[a])
       // {
            b+=3;
       // }
        m[c] = m[a];
        m[a] = min;
        prim (m,n);
        if(j < n -2)
        {
            printf("\n");
        }
        a++;
        min=INT_MAX;
    }
    s=s-7;
    return 0;
}
