#include <stdio.h>

long int F(int k);

int main(void) {

    int k;

    scanf("%d", &k);
    printf("%ld\n", F(k));

    return 0;

}

long int F(int k) {
    
    if (k<1) /* условие останова */

        return 0;
        
    else if (k==1)
        
        return 1;
        
    else if (k==2)
        
        return 1;

    else if (k>2)

        return (F(k-1) + F(k-2));  /* рекурсивный вызов */

}
