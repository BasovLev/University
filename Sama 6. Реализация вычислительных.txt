#include <stdio.h>
#include <math.h>

int main()
{
    double a, b, c, d, x, x0, n;
    int k;
    a = 6.0;
    b = 8.0;
    d = 1.0;
    c=0;
    k=0;
    n=0.0;
    while (fabs(b-a)>1*(1e-7)){
    
        x=(a+b)/2.0;
        
        if (cos(x)<(3.0/5)){
            a=x;
        }
         if (cos(x)>=(3.0/5)){
            b=x;
        }
        
    }
    c = (fabs(cos(b) - cos(a)));
    x0=a;
    while (fabs(n - cos(x0)) > 1*(1e-6))
    {
        k+=1;
        n+=d;
        d = d/((2.0*k - 1.0)*(2.0*k));
        d = d*x0*x0*(-1.0);
    }
    printf("%.7f\n", x0);
    printf("%.7f\n", c);
    printf("%.6f\n", n);
    printf("%d\n", (k-1));
}
